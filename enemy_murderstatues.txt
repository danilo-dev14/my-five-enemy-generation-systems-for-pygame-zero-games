import random

# Definicoes da tela
WIDTH = 800
HEIGHT = 600

# Lista para armazenar os inimigos (Actors)
enemies = []

# Numero maximo de inimigos (garante que nao tera mais que 10)
MAX_ENEMIES = 10 

# --- üöÄ Fun√ß√µes Principais ---

def create_enemies(count):
    """
    Gera o numero especificado de inimigos (no maximo MAX_ENEMIES) 
    em posicoes aleatorias.
    """
    # Limita a contagem para garantir que nao ultrapasse o maximo
    if count > MAX_ENEMIES:
        count = MAX_ENEMIES
    
    for _ in range(count):
        # Cria um Actor. A string 'enemy' e o nome do arquivo de imagem.
        # CERTIFIQUE-SE DE TER UM ARQUIVO CHAMADO 'enemy.png' na pasta 'images'
        enemy = Actor('enemy') 
        
        # Define uma posicao x e y aleatoria dentro dos limites da tela
        # Os inimigos serao IM√ìVEIS, pois nao ha codigo para atualiza-los (update)
        enemy.x = random.randint(enemy.width // 2, WIDTH - enemy.width // 2)
        enemy.y = random.randint(enemy.height // 2, HEIGHT - enemy.height // 2)
        
        enemies.append(enemy)

# --- üñºÔ∏è Fun√ß√µes de Desenho e Atualiza√ß√£o ---

    Funcao de desenho chamada automaticamente pelo Pygame Zero para
